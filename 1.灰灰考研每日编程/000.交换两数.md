#include <stdio.h>
#define MAX 1000//数组最大长度

// 异或交换
void swap(int *a, int *b);
//中间变量交换
void swap(int *a, int *b);
//数组交换交换
void swap(int arr[], int i, int j);

int main() {
    int x=2, y=5;
    // 调用swap函数
    swap(&x, &y);
    return 0;
}

// 使用异或运算交换两个数，&x != &y，即 a != b，若是同一个地址则会置为0
void swap(int *a, int *b) {
    if (*a != *b && a != b) { // 确保两个数不相同，避免无限循环
        *a = *a ^ *b; // 交换a和b的值
        *b = *a ^ *b; // 恢复a的原始值
        *a = *a ^ *b; // 将b的原始值赋给a
    }
}
void swap(int *a, int *b);{
	int temp = 0;
	temp = *a;
	*a = *b;
	*b = temp;
}
//数组交换交换
void swap(int arr[], int i, int j);{
    if(i >= 0 && i < MAX && j >= 0 && j < MAX){
    	//调用函数
    	swap(&arr[i],&arr[j]);
    }
}
void swap(int arr[], int i, int j);{
	int temp = 0;
	if(i >= 0 && i < MAX && j >= 0 && j < MAX){
		//中间变量
    	temp = arr[i];
    	arr[i] = arr[j];
    	arr[j] = temp;
    }
}
void swap(int arr[], int i, int j);{
	if(i >= 0 && i < MAX && j >= 0 && j < MAX){
		//异或运算
    	arr[i] = arr[i] ^ arr[j];
    	arr[j] = arr[i] ^ arr[j];
    	arr[i] = arr[i] ^ arr[j];
    }
}
